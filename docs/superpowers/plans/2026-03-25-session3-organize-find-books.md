```sql
-- FTS5 virtual table for full-text search
CREATE VIRTUAL TABLE IF NOT EXISTS books_fts USING fts5(
  title,
  author,
  description,
  content='books',
  content_rowid='rowid'
);

-- Populate FTS from existing books
INSERT INTO books_fts(rowid, title, author, description)
SELECT rowid, title, author, COALESCE(description, '') FROM books;

-- Keep FTS in sync with books table
CREATE TRIGGER books_fts_insert AFTER INSERT ON books BEGIN
  INSERT INTO books_fts(rowid, title, author, description)
  VALUES (NEW.rowid, NEW.title, NEW.author, COALESCE(NEW.description, ''));
END;

CREATE TRIGGER books_fts_delete AFTER DELETE ON books BEGIN
  INSERT INTO books_fts(books_fts, rowid, title, author, description)
  VALUES ('delete', OLD.rowid, OLD.title, OLD.author, COALESCE(OLD.description, ''));
END;

CREATE TRIGGER books_fts_update AFTER UPDATE ON books BEGIN
  INSERT INTO books_fts(books_fts, rowid, title, author, description)
  VALUES ('delete', OLD.rowid, OLD.title, OLD.author, COALESCE(OLD.description, ''));
  INSERT INTO books_fts(rowid, title, author, description)
  VALUES (NEW.rowid, NEW.title, NEW.author, COALESCE(NEW.description, ''));
END;
```

## Task 4: buildFilterConditions utility

**Files:**
- Create: `packages/server/src/trpc/routers/build-filter.ts`

- [ ] **Step 1: Write the filter builder**

```typescript
import { eq, ne, gt, gte, lt, lte, like, and, or, sql } from "drizzle-orm";
import { books } from "@verso/shared";
import type { SmartFilter, SmartFilterCondition } from "@verso/shared";

function escapeLike(str: string): string {
  return str.replace(/[%_\\]/g, (ch) => `\\${ch}`);
}

const columnMap = {
  title: books.title,
  author: books.author,
  genre: books.genre,
  year: books.year,
  language: books.language,
  fileFormat: books.fileFormat,
  pageCount: books.pageCount,
} as const;

function buildCondition(cond: SmartFilterCondition) {
  const { field, op, value } = cond;

  // Tags require special handling — stored as JSON array
  if (field === "tags") {
    const strVal = String(value);
    switch (op) {
      case "eq":
      case "contains":
        return sql`EXISTS (SELECT 1 FROM json_each(${books.tags}) WHERE value = ${strVal})`;
      case "neq":
        return sql`NOT EXISTS (SELECT 1 FROM json_each(${books.tags}) WHERE value = ${strVal})`;
      case "in":
        if (Array.isArray(value)) {
          const conditions = value.map(
            (v) => sql`EXISTS (SELECT 1 FROM json_each(${books.tags}) WHERE value = ${v})`
          );
          return or(...conditions);
        }
        return sql`1=0`;
      default:
        return sql`1=0`;
    }
  }

  const column = columnMap[field as keyof typeof columnMap];
  if (!column) return sql`1=0`;

  switch (op) {
    case "eq": return eq(column, value as any);
    case "neq": return ne(column, value as any);
    case "contains": return like(column, `%${escapeLike(String(value))}%`);
    case "gt": return gt(column, value as any);
    case "gte": return gte(column, value as any);
    case "lt": return lt(column, value as any);
    case "lte": return lte(column, value as any);
    case "in":
      if (Array.isArray(value)) {
        return or(...value.map((v) => eq(column, v as any)));
      }
      return sql`1=0`;
    default: return sql`1=0`;
  }
}

export function buildFilterConditions(filter: SmartFilter) {
  const conditions = filter.conditions.map(buildCondition);
  if (filter.operator === "OR") {
    return or(...conditions);
  }
  return and(...conditions);
}
```

```typescript
const DEFAULT_SHELVES = [
  { name: "Currently Reading", emoji: "📖", isSmart: false, isDefault: true, position: 0 },
  { name: "Want to Read", emoji: "🔖", isSmart: false, isDefault: true, position: 1 },

    isSmart: true,
    isDefault: true,
    position: 3,
    // Special sentinel: evaluated as "books added in last 30 days" in the router,
    // not through the generic filter builder.
    smartFilter: JSON.stringify({
      operator: "AND",
      conditions: [{ field: "_recentlyAdded", op: "lte", value: "30" }],
    }),
  },
];

export async function seedDefaultShelves(db: AppDatabase, userId: string) {
  for (const shelf of DEFAULT_SHELVES) {
    await db.insert(shelves).values({
      ...shelf,
      userId,
    });
  }
}
```

```typescript
      .where(eq(shelves.userId, ctx.user.sub))
      .orderBy(shelves.position);

    // Get book counts for manual shelves
    const counts = await ctx.db
      .select({
        shelfId: shelfBooks.shelfId,
        count: sql<number>`count(*)`,
      })
      .from(shelfBooks)
      .groupBy(shelfBooks.shelfId);

    const countMap = new Map(counts.map((c) => [c.shelfId, c.count]));

    return userShelves.map((shelf) => ({
      ...shelf,
      bookCount: shelf.isSmart ? null : (countMap.get(shelf.id) ?? 0),
    }));
  }),

  byId: protectedProcedure.input(shelfByIdInput).query(async ({ ctx, input }) => {
    const shelf = await ctx.db.query.shelves.findFirst({
      where: and(eq(shelves.id, input.id), eq(shelves.userId, ctx.user.sub)),
    });
    if (!shelf) throw new TRPCError({ code: "NOT_FOUND", message: "Shelf not found" });

    let shelfBooksList: typeof books.$inferSelect[];

    if (shelf.isSmart && shelf.smartFilter) {
      const filter = JSON.parse(shelf.smartFilter) as SmartFilter;
      const userCondition = eq(books.addedBy, ctx.user.sub);

      // Special-case: "_recentlyAdded" sentinel means "books added in last N days"
      const hasRecentlyAdded = filter.conditions.some((c) => c.field === "_recentlyAdded");
      if (hasRecentlyAdded) {
        const days = filter.conditions.find((c) => c.field === "_recentlyAdded")?.value ?? 30;
        shelfBooksList = await ctx.db
          .select()
          .from(books)
          .where(and(
            userCondition,
            sql`${books.createdAt} >= datetime('now', ${`-${days} days`})`,
          ))
          .orderBy(desc(books.createdAt));
      } else {
        const filterWhere = buildFilterConditions(filter);
        shelfBooksList = await ctx.db
          .select()
          .from(books)
          .where(and(userCondition, filterWhere))
          .orderBy(desc(books.createdAt));
      }
    } else {
      // Manual shelf: join through shelfBooks, select full book record
      shelfBooksList = await ctx.db
        .select({
          id: books.id,
          title: books.title,
          author: books.author,
          isbn: books.isbn,
          publisher: books.publisher,
          year: books.year,
          language: books.language,
          description: books.description,
          genre: books.genre,
          tags: books.tags,
          coverPath: books.coverPath,
          filePath: books.filePath,
          fileFormat: books.fileFormat,
          fileSize: books.fileSize,
          fileHash: books.fileHash,
          pageCount: books.pageCount,
          addedBy: books.addedBy,
          metadataSource: books.metadataSource,
          metadataLocked: books.metadataLocked,
          createdAt: books.createdAt,
          updatedAt: books.updatedAt,
        })
        .from(shelfBooks)
        .innerJoin(books, eq(books.id, shelfBooks.bookId))
        .where(eq(shelfBooks.shelfId, shelf.id))
        .orderBy(shelfBooks.position);
    }

    return { shelf, books: shelfBooksList };
  }),

  create: protectedProcedure.input(shelfCreateInput).mutation(async ({ ctx, input }) => {
    // Get next position
    const maxPos = ctx.db
      .select({ max: sql<number>`COALESCE(MAX(${shelves.position}), -1)` })
      .from(shelves)
      .where(eq(shelves.userId, ctx.user.sub))
      .get();

    const [shelf] = await ctx.db
      .insert(shelves)
      .values({
        name: input.name,
        emoji: input.emoji,
        description: input.description,
        isSmart: input.isSmart,
        smartFilter: input.smartFilter ? JSON.stringify(input.smartFilter) : null,
        userId: ctx.user.sub,
        position: (maxPos?.max ?? -1) + 1,
      })
      .returning();

    return shelf;
  }),

  update: protectedProcedure.input(shelfUpdateInput).mutation(async ({ ctx, input }) => {
    const existing = await ctx.db.query.shelves.findFirst({
      where: and(eq(shelves.id, input.id), eq(shelves.userId, ctx.user.sub)),
    });
    if (!existing) throw new TRPCError({ code: "NOT_FOUND", message: "Shelf not found" });

    const updateData: Record<string, any> = { ...timestamp() };
    if (input.name !== undefined) updateData.name = input.name;
    if (input.emoji !== undefined) updateData.emoji = input.emoji;
    if (input.description !== undefined) updateData.description = input.description;
    if (input.smartFilter !== undefined) updateData.smartFilter = JSON.stringify(input.smartFilter);

    const [shelf] = await ctx.db
      .update(shelves)
      .set(updateData)
      .where(eq(shelves.id, input.id))
      .returning();

    return shelf;
  }),

  delete: protectedProcedure.input(shelfByIdInput).mutation(async ({ ctx, input }) => {
    const existing = await ctx.db.query.shelves.findFirst({
      where: and(eq(shelves.id, input.id), eq(shelves.userId, ctx.user.sub)),
    });
    if (!existing) throw new TRPCError({ code: "NOT_FOUND", message: "Shelf not found" });

    await ctx.db.delete(shelves).where(eq(shelves.id, input.id));
    return { success: true };
  }),

  reorder: protectedProcedure.input(shelfReorderInput).mutation(async ({ ctx, input }) => {
    for (let i = 0; i < input.shelfIds.length; i++) {
      await ctx.db
        .update(shelves)
        .set({ position: i, ...timestamp() })
        .where(and(eq(shelves.id, input.shelfIds[i]), eq(shelves.userId, ctx.user.sub)));
    }
    return { success: true };
  }),

  addBook: protectedProcedure.input(shelfBookInput).mutation(async ({ ctx, input }) => {
    const shelf = await ctx.db.query.shelves.findFirst({
      where: and(eq(shelves.id, input.shelfId), eq(shelves.userId, ctx.user.sub)),
    });
    if (!shelf) throw new TRPCError({ code: "NOT_FOUND", message: "Shelf not found" });
    if (shelf.isSmart) throw new TRPCError({ code: "BAD_REQUEST", message: "Cannot add books to smart shelves" });

    const maxPos = ctx.db
      .select({ max: sql<number>`COALESCE(MAX(${shelfBooks.position}), -1)` })
      .from(shelfBooks)
      .where(eq(shelfBooks.shelfId, input.shelfId))
      .get();

    await ctx.db.insert(shelfBooks).values({
      shelfId: input.shelfId,
      bookId: input.bookId,
      position: (maxPos?.max ?? -1) + 1,
    });

    return { success: true };
  }),

  removeBook: protectedProcedure.input(shelfBookInput).mutation(async ({ ctx, input }) => {
    await ctx.db
      .delete(shelfBooks)
      .where(and(eq(shelfBooks.shelfId, input.shelfId), eq(shelfBooks.bookId, input.bookId)));
    return { success: true };
  }),
});
```

```typescript
  search: protectedProcedure.input(searchInput).query(async ({ ctx, input }) => {
    const { query, genre, author, format, page, limit } = input;
    const offset = (page - 1) * limit;

    // FTS5 search with bm25 ranking (title=10, author=5, description=1)
    // Use Drizzle's sql template + db.all() from the underlying better-sqlite3.
    // The Drizzle better-sqlite3 adapter stores the raw connection at (ctx.db as any)._.session.client
    // Alternatively, expose the raw sqlite instance from db/client.ts.
    //
    // Preferred approach: add a `rawDb` field to createDb() return or use Drizzle's
    // `db.execute()` which returns raw rows. Test which works and adapt.
    //
    // The FTS5 content table syncs via rowid. SQLite tables with text PKs still have
    // implicit rowids. Verify: SELECT rowid, id FROM books LIMIT 1; should return both.

    const ftsQuery = sql`
      SELECT b.*, bm25(books_fts, 10, 5, 1) AS rank
      FROM books_fts
      JOIN books b ON b.rowid = books_fts.rowid
      WHERE books_fts MATCH ${query}
        AND b.added_by = ${ctx.user.sub}
        ${genre ? sql`AND b.genre = ${genre}` : sql``}
        ${author ? sql`AND b.author LIKE ${"%" + escapeLike(author) + "%"} ESCAPE '\\'` : sql``}
        ${format ? sql`AND b.file_format = ${format}` : sql``}
      ORDER BY rank
      LIMIT ${limit} OFFSET ${offset}
    `;

    const countQuery = sql`
      SELECT count(*) as total
      FROM books_fts
      JOIN books b ON b.rowid = books_fts.rowid
      WHERE books_fts MATCH ${query}
        AND b.added_by = ${ctx.user.sub}
        ${genre ? sql`AND b.genre = ${genre}` : sql``}
        ${author ? sql`AND b.author LIKE ${"%" + escapeLike(author) + "%"} ESCAPE '\\'` : sql``}
        ${format ? sql`AND b.file_format = ${format}` : sql``}
    `;

    // Drizzle better-sqlite3: use .all() and .get() on the raw session
    // Access pattern: (ctx.db as any)._.session.client gives the BetterSqlite3.Database
    // Then: rawDb.prepare(compiledQuery.sql).all(...compiledQuery.params)
    //
    // Simpler alternative that works with Drizzle v0.39+:
    const ftsResults = ctx.db.all(ftsQuery) as any[];
    const countResult = ctx.db.get(countQuery) as any;

    return {
      books: ftsResults,
      total: countResult?.total ?? 0,
      page,
    };
  }),
```

**Important:** The Drizzle `better-sqlite3` adapter exposes `.all(sql)` and `.get(sql)` for raw SQL queries. If these don't exist on your version, access the raw connection: `const rawDb = (ctx.db as any)._.session.client as BetterSqlite3.Database;` then use `rawDb.prepare(query.sql).all(...query.params)`. The implementer should verify `SELECT rowid, id FROM books LIMIT 1` works in a test before proceeding — this confirms FTS5 content sync will work correctly.

```typescript
import type { SmartFilter, SmartFilterCondition } from "@verso/shared";

const FIELDS = [
  { value: "title", label: "Title" },
  { value: "author", label: "Author" },
  { value: "genre", label: "Genre" },
  { value: "tags", label: "Tags" },
  { value: "year", label: "Year" },
  { value: "language", label: "Language" },
  { value: "fileFormat", label: "Format" },
  { value: "pageCount", label: "Page Count" },
];

const OPS = [
  { value: "eq", label: "equals" },
  { value: "neq", label: "not equals" },
  { value: "contains", label: "contains" },
  { value: "gt", label: "greater than" },
  { value: "gte", label: "at least" },
  { value: "lt", label: "less than" },
  { value: "lte", label: "at most" },
];

type FilterBuilderProps = {
  filter: SmartFilter;
  onChange: (filter: SmartFilter) => void;
};

export function FilterBuilder({ filter, onChange }: FilterBuilderProps) {
  const updateCondition = (index: number, updates: Partial<SmartFilterCondition>) => {
    const conditions = [...filter.conditions];
    conditions[index] = { ...conditions[index], ...updates } as SmartFilterCondition;
    onChange({ ...filter, conditions });
  };

  const addCondition = () => {
    onChange({
      ...filter,
      conditions: [...filter.conditions, { field: "genre", op: "eq", value: "" }],
    });
  };

  const removeCondition = (index: number) => {
    if (filter.conditions.length <= 1) return;
    onChange({
      ...filter,
      conditions: filter.conditions.filter((_, i) => i !== index),
    });
  };

  return (
    <div className="space-y-3">
      <div className="flex items-center gap-2">
        <span className="text-xs" style={{ color: "var(--text-dim)" }}>Match</span>
        <select
          value={filter.operator}
          onChange={(e) => onChange({ ...filter, operator: e.target.value as "AND" | "OR" })}
          className="rounded-lg border px-2 py-1 text-xs"
          style={{ backgroundColor: "var(--card)", borderColor: "var(--border)", color: "var(--text)" }}
        >
          <option value="AND">all conditions</option>
          <option value="OR">any condition</option>
        </select>
      </div>

      {filter.conditions.map((cond, i) => (
        <div key={i} className="flex items-center gap-2">
          <select
            value={cond.field}
            onChange={(e) => updateCondition(i, { field: e.target.value as any })}
            className="rounded-lg border px-2 py-1.5 text-xs flex-1"
            style={{ backgroundColor: "var(--card)", borderColor: "var(--border)", color: "var(--text)" }}
          >
            {FIELDS.map((f) => <option key={f.value} value={f.value}>{f.label}</option>)}
          </select>

          <select
            value={cond.op}
            onChange={(e) => updateCondition(i, { op: e.target.value as any })}
            className="rounded-lg border px-2 py-1.5 text-xs"
            style={{ backgroundColor: "var(--card)", borderColor: "var(--border)", color: "var(--text)" }}
          >
            {OPS.map((o) => <option key={o.value} value={o.value}>{o.label}</option>)}
          </select>

          <input
            type="text"
            value={String(cond.value)}
            onChange={(e) => {
              const val = ["year", "pageCount"].includes(cond.field)
                ? Number(e.target.value) || 0
                : e.target.value;
              updateCondition(i, { value: val });
```
